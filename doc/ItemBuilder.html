def images(images)
	@item.images = images.map do |image|
		Attachment::Image.new(
			x: image[:x],
			y: image[:y],
			w: image[:w],
			h: image[:h],
			main: image[:main] ) do |i|
				i.id=image[:id]
			end
	end
	self
end
